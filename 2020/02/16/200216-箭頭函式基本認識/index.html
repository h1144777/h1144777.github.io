1.(參數列表)=>(回傳值)

//傳統的函式宣告
let add=function(n1+n2){
return n1+n2;
};

//箭頭函式的寫法1
let add=(n1,n2)=>{
return n1+n2;
};

2.(參數列表)=>{函式內部程式}

//傳統的函示宣告
let add=function(n1+n2){
return n1+n2;
};

//箭頭函式的寫法2
let add=(n1,n2)=>{
return n1+n2;
};

匿名函式:

//在排程中使用”傳統方式”撰寫”匿名函式”
setTimeout(function(){
console.log(“過了一秒”);
},1000);

//在排程中使用”箭頭函式”，撰寫”匿名函式”
setTimeout(()=>{
console.log(“過了一秒”);
},1000);
